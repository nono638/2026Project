### `src/chunkers/semantic.py`

Wraps LangChain's SemanticChunker. Logic comes from `src/pipeline/chunking.py`.

```python
from langchain_experimental.text_splitter import SemanticChunker as LCSemanticChunker
from langchain_community.embeddings import OllamaEmbeddings


class SemanticChunker:
    """Semantic chunking using LangChain's SemanticChunker.

    Uses an embedding model to find meaning boundaries in text.
    Note: the embedding model used for chunking is independent of the
    embedding model used for retrieval. They serve different purposes —
    chunking embeddings find breakpoints, retrieval embeddings find similarity.
    """

    def __init__(self, embedding_model: str = "mxbai-embed-large"):
        self._model = embedding_model

    @property
    def name(self) -> str:
        return f"semantic:{self._model}"

    def chunk(self, text: str) -> list[str]:
        embeddings = OllamaEmbeddings(model=self._model)
        chunker = LCSemanticChunker(embeddings)
        docs = chunker.create_documents([text])
        return [doc.page_content for doc in docs]
```

### `src/embedders/ollama.py`

Logic comes from `src/pipeline/retrieval.py` `embed_texts()`.

```python
import numpy as np
from ollama import Client


class OllamaEmbedder:
    """Embedding via any Ollama-hosted embedding model."""

    def __init__(self, model: str = "mxbai-embed-large"):
        self._model = model
        self._client = Client()
        self._dimension: int | None = None

    @property
    def name(self) -> str:
        return f"ollama:{self._model}"

    @property
    def dimension(self) -> int:
        if self._dimension is None:
            # Detect dimension by embedding a probe string
            probe = self.embed(["hello"])
            self._dimension = probe.shape[1]
        return self._dimension

    def embed(self, texts: list[str]) -> np.ndarray:
        response = self._client.embed(model=self._model, input=texts)
        result = np.array(response.embeddings, dtype=np.float32)
        if self._dimension is None:
            self._dimension = result.shape[1]
        return result
```

### `src/scorers/claude.py`

Logic comes from `src/pipeline/scoring.py`.

```python
import anthropic


JUDGE_PROMPT = """You are evaluating the quality of an answer generated by a language model.

**Question:** {query}

**Reference context (ground truth source):**
{context}

**Model's answer:**
{answer}

Rate the answer on three dimensions, each from 1 (worst) to 5 (best):

1. **Faithfulness**: Is the answer factually consistent with the context? (1 = contradicts context, 5 = perfectly faithful)
2. **Relevance**: Does the answer address the question asked? (1 = completely off-topic, 5 = directly answers the question)
3. **Conciseness**: Is the answer appropriately concise without losing important information? (1 = extremely verbose or missing key info, 5 = perfectly concise)

Respond in exactly this format:
faithfulness: <score>
relevance: <score>
conciseness: <score>"""


class ClaudeScorer:
    """LLM-as-judge scoring using Claude.

    Methodology: Saad-Falcon et al. (2024, NAACL) showed LLM-as-judge
    correlates well with human judgment.

    Known limitation: if Claude is also used for query classification,
    both features and labels carry Claude's biases.
    """

    def __init__(self, model: str = "claude-sonnet-4-20250514"):
        self._model = model
        self._client = anthropic.Anthropic()

    @property
    def name(self) -> str:
        return f"claude:{self._model}"

    def score(self, query: str, context: str, answer: str) -> dict[str, float]:
        response = self._client.messages.create(
            model=self._model,
            max_tokens=100,
            messages=[{
                "role": "user",
                "content": JUDGE_PROMPT.format(
                    query=query, context=context, answer=answer
                ),
            }],
        )
        text = response.content[0].text
        scores = {}
        for line in text.strip().split("\n"):
            line = line.strip()
            for dimension in ("faithfulness", "relevance", "conciseness"):
                if line.lower().startswith(dimension):
                    try:
                        scores[dimension] = float(line.split(":")[-1].strip())
                    except ValueError:
                        scores[dimension] = 0.0
        return scores
```

### `src/strategies/naive.py`

Logic from `src/pipeline/strategies/naive.py`. Key change: receives `Retriever` instead
of `(chunks, index)`.

```python
from ollama import Client


class NaiveRAG:
    """Naive RAG — the baseline.

    Query -> retrieve top-k chunks -> generate answer.
    No special logic. Raw model capability is the only variable.
    """

    def __init__(self):
        self._client = Client()

    @property
    def name(self) -> str:
        return "naive"

    def run(self, query: str, retriever, model: str) -> str:
        retrieved = retriever.retrieve(query)
        context = "\n\n".join(r["text"] for r in retrieved)

        prompt = (
            f"Answer the following question using only the provided context.\n\n"
            f"Context:\n{context}\n\n"
            f"Question: {query}\n\n"
            f"Answer:"
        )

        response = self._client.chat(
            model=model,
            messages=[{"role": "user", "content": prompt}],
        )
        return response.message.content
```
